---
title: "[WIP] Get Request Status"
api: "GET https://api-preview.isomeric.ai/request-status/:id"
description: "This endpoint returns the status of an `/extract` request"
---

### Headers

<ParamField
  header="Authorization"
  required="true"
  initialValue="Bearer iso_abc123"
>
  Use your API key to authenticate the request.
</ParamField>

### Params

<ParamField body=":id" type="string" required="true">
  The request (job) ID returned from your call to `/extract`
</ParamField>

### Response

<ResponseField name="credits" type="object">
  Info about API credit usage for this request.

  <Expandable title="Toggle object">
    <ResponseField name="remaining" type="number">
      The number of credits remaining after this call
    </ResponseField>
  </Expandable>
</ResponseField>
